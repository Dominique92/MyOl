This package defines :
  • a tile layers collection
  • and a layer switcher control

Include standard Openlayers distribution ol.css & ol.js
Include this package tileLayers.css & tileLayers.js
	new ol.Map({
		target: 'map',
		controls: [
			new ol.control.Zoom(),
			controlLayerSwitcher({
				baseLayers: layersCollection(),
				addonEl: document.getElementById('addon'),
			}),
		],
		view: new ol.View({
			center: [0, 0],
			zoom: 2,
		}),
	});
